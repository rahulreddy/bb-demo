# create bucket
aws s3api create-bucket --bucket demo --endpoint http://localhost:8000

# enable versioning
aws s3api put-bucket-versioning --bucket demo --versioning-configuration Status=Enabled --endpoint http://localhost:8000

# enable replication
aws s3api put-bucket-replication --bucket demo --replication-configuration  file://replication.json --endpoint http://localhost:8000

# put object
aws s3api put-object --bucket demo --key foo --body ./TESTDATA --endpoint http://localhost:8000

# head object
aws s3api head-object --bucket demo --key foo --endpoint http://localhost:8000

# list objects
watch -n1 aws s3api list-objects --bucket demo --endpoint http://localhost:8000
